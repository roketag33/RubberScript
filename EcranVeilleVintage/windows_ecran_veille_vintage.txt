REM ===========================================
REM Écran de Veille Vintage - Windows
REM ===========================================
REM Active des screensavers vintage Windows
REM Utilise les .scr natifs du système
REM Nettoie automatiquement les traces
REM Compatible Windows 10/11

REM ---- CONFIGURATION ----
REM Screensavers : Intégrés Windows
REM Durée : 30 secondes par screensaver
REM Nombre : 5 screensavers différents
REM Sortie : Mouvement de souris automatique

DELAY 1000

REM ========== SCREENSAVER 1 - BULLES ==========
REM Active l'écran de veille Bulles
GUI r
DELAY 300
STRING C:\Windows\System32\Bubbles.scr /s
ENTER
DELAY 30000
REM Sortie par mouvement de souris
MOUSEMOVE 100 100
DELAY 1000

REM ========== CONFIGURATION SCREENSAVER ==========
REM Ouvre les paramètres d'écran de veille
GUI r
DELAY 300
STRING control desk.cpl,,1
ENTER
DELAY 5000
REM Ferme les paramètres
ALT F4
DELAY 1000

REM ========== NETTOYAGE ==========
REM Efface les traces dans l'historique Exécuter
GUI r
DELAY 300
STRING regedit
ENTER
DELAY 2000
REM Ferme regedit sans modification
ALT F4
DELAY 1000

REM ========== RESTAURATION ==========
REM Remet l'écran de veille par défaut
GUI r
DELAY 300
STRING control desk.cpl,,1
ENTER
DELAY 3000
REM Sélectionne "Aucun"
TAB
TAB
TAB
DOWN
UP
ENTER
DELAY 1000
REM Applique et ferme
TAB
TAB
TAB
TAB
ENTER
DELAY 1000
