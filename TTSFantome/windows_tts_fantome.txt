REM ======================================
REM TTS Fantôme - Windows PowerShell
REM ======================================
REM Utilise le moteur de synthèse vocale Windows
REM Phrases loufoques à intervalles réguliers
REM Volume modéré pour éviter les problèmes
REM Nettoie automatiquement les traces

REM ---- CONFIGURATION ----
REM Nombre de phrases : 5 phrases différentes
REM Intervalle : 10-30 secondes entre chaque phrase
REM Volume : 50% (discret mais audible)

DELAY 1000

REM ========== PHRASE 1 ==========
REM Ouvre PowerShell en arrière-plan
GUI r
DELAY 300
STRING powershell -WindowStyle Hidden -Command "Add-Type -AssemblyName System.Speech; $synth = New-Object System.Speech.Synthesis.SpeechSynthesizer; $synth.Volume = 50; $synth.Speak('Bonjour, je suis votre ordinateur et je me sens un peu seul aujourd hui')"
